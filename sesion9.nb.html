library(ggplot2)

Grfico de barras

ggplot(data_barras, aes(x = nombre, y = valor)) +
    geom_bar(
        stat = "identity",
        color = "blue", # Algunos colores bsicos se pueden llamar por nombre
        fill = rgb(0.1, 0.4, 0.5, 0.7) # Para otros colores usamos su cdigo
    )

ggplot(data_barras, aes(x = nombre, y = valor, fill = nombre)) +
    geom_bar(stat = "identity") +
    scale_fill_brewer(palette = "Set1") +
    labs(
        title = "Grfico de barras",
        x = "Departamento",
        y = "Edad promedio"
    )
